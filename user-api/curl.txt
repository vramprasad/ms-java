# Healthcheck
curl http://localhost:9901/user-api/v1/users/healthcheck

# Get all items
curl http://localhost:9901/user-api/v1/users/listall

# Create an item
curl -X POST -H "Content-Type: application/json" -d '{"name":"Test Name","email":"test@email.com"}' http://localhost:9901/user-api/v1/users/

# Get a specific item
curl http://localhost:9901/user-api/v1/users/1

# Update an item
curl -X PUT -H "Content-Type: application/json" -d '{"name":"Updated Name","email":"updated@email.com"}' http://localhost:9901/user-api/v1/users/2

# Delete an item
curl -X DELETE http://localhost:9901/user-api/v1/users/1
